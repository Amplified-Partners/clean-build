# OVERVIEW

## What This Does

Every Covered AI customer automatically gets optimised for AI search engines. No effort required from the customer — it just happens.

**The system:**
1. Generates an AI-optimised public page for each business
2. Creates FAQ content from actual call patterns
3. Injects comprehensive schema markup for machine readability
4. Monitors AI platforms for business mentions
5. Reports visibility score in dashboard

**Why it matters:**
- AI engines (ChatGPT, Perplexity, Google AI Overviews) are replacing traditional search
- Businesses that are structured for AI citation get recommended
- This is a genuine differentiator — competitors don't do this automatically

## Job 1: Generate GEO Page

### File
`src/trigger/geo/generate-geo-page.ts`

### Trigger
- Cron: Daily at 3am
- Event: On significant data change (new reviews, 10+ new calls)

### Implementation

```typescript
import { task, schedules } from "@trigger.dev/sdk/v3";
import { prisma } from "@/lib/prisma";
import { generateFaqsFromCalls } from "@/lib/geo/faq-generator";
import { extractServicesFromJobs } from "@/lib/geo/service-extractor";
import { generateSchemaMarkup } from "@/lib/geo/schema-generator";
import { slugify } from "@/lib/utils";

export const generateGeoPage = schedules.task({
  id: "generate-geo-page",
  cron: "0 3 * * *", // 3am daily
  run: async () => {
    const clients = await prisma.client.findMany({
      where: { status: 'ACTIVE' },
      include: {
        owner: true,
        geoPage: true,
        jobs: {
          where: { status: 'COMPLETED' },
          take: 100,
          orderBy: { completedAt: 'desc' }
        },
        calls: {
          where: { 
            createdAt: { gte: new Date(Date.now() - 90 * 24 * 60 * 60 * 1000) } // 90 days
          },
          orderBy: { createdAt: 'desc' }
        },
        reviews: {
          where: { status: 'COMPLETED', rating: { gte: 4 } },
          take: 20,
          orderBy: { createdAt: 'desc' }
        }
      }
    });
    
    let generated = 0;
    
    for (const client of clients) {
      try {
        // 1. Generate FAQs from call transcripts
        const faqs = await generateFaqsFromCalls(client.id, client.calls);
        
        // 2. Extract services from job history
        const services = await extractServicesFromJobs(client.jobs, client.vertical);
        
        // 3. Build comprehensive schema markup
        const schemaMarkup = generateSchemaMarkup({
          client,
          faqs: faqs.filter(f => f.isPublished),
          services,
          reviews: client.reviews
        });
        
        // 4. Generate slug if new
        const slug = client.geoPage?.slug || slugify(client.businessName);
        
        // 5. Upsert GEO page
        await prisma.geoPage.upsert({
          where: { clientId: client.id },
          create: {
            clientId: client.id,
            slug,
            businessDescription: client.geoPage?.businessDescription || 
              generateDefaultDescription(client),
            serviceAreas: client.serviceAreas || [],
            serviceAreaText: client.serviceAreas?.join(', ') || '',
            services,
            faqs: faqs.map(f => ({
              question: f.question,
              answer: f.answer,
              category: f.category
            })),
            schemaMarkup,
            lastGenerated: new Date()
          },
          update: {
            services,
            faqs: faqs.map(f => ({
              question: f.question,
              answer: f.answer,
              category: f.category
            })),
            schemaMarkup,
            lastGenerated: new Date()
          }
        });
        
        generated++;
      } catch (error) {
        console.error(`Failed to generate GEO page for ${client.id}:`, error);
      }
    }
    
    return { generated, total: clients.length };
  }
});
```

## Job 2: Extract FAQs from Calls

### File
`src/trigger/geo/extract-faqs.ts`

### Trigger
- Cron: Weekly on Sunday at 2am
- Event: When 20+ new calls accumulated

### Implementation

```typescript
import { task, schedules } from "@trigger.dev/sdk/v3";
import { prisma } from "@/lib/prisma";
import OpenAI from "openai";

const openai = new OpenAI();

interface ExtractedFaq {
  question: string;
  answer: string;
  category: string;
  confidence: number;
}

export const extractFaqsFromCalls = schedules.task({
  id: "extract-faqs-from-calls",
  cron: "0 2 * * 0", // Sunday 2am
  run: async () => {
    const clients = await prisma.client.findMany({
      where: { status: 'ACTIVE' },
      include: {
        calls: {
          where: {
            createdAt: { gte: new Date(Date.now() - 30 * 24 * 60 * 60 * 1000) },
            transcript: { not: null }
          },
          select: {
            id: true,
            transcript: true,
            summary: true,
            intent: true
          }
        },
        generatedFaqs: {
          where: { isEdited: false } // Don't overwrite edited FAQs
        }
      }
    });
    
    let totalFaqs = 0;
    
    for (const client of clients) {
      if (client.calls.length < 5) continue; // Need minimum data
      
      const faqs = await generateFaqsWithAI(client.calls, client.vertical);
      
      // Upsert FAQs - merge with existing, don't duplicate
      for (const faq of faqs) {
        const existing = client.generatedFaqs.find(
          f => similarQuestion(f.question, faq.question)
        );
        
        if (existing) {
          // Update frequency and confidence if not manually edited
          if (!existing.isEdited) {
            await prisma.generatedFaq.update({
              where: { id: existing.id },
              data: {
                frequency: { increment: 1 },
                confidence: Math.max(existing.confidence, faq.confidence),
                answer: faq.confidence > existing.confidence ? faq.answer : existing.answer
              }
            });
          }
        } else {
          // Create new FAQ
          await prisma.generatedFaq.create({
            data: {
              clientId: client.id,
              question: faq.question,
              answer: faq.answer,
              category: faq.category,
              confidence: faq.confidence,
              sourceCallIds: faq.sourceCallIds || []
            }
          });
          totalFaqs++;
        }
      }
    }
    
    return { faqsGenerated: totalFaqs };
  }
});

async function generateFaqsWithAI(
  calls: Array<{ id: string; transcript: string; summary: string; intent: string }>,
  vertical: string
): Promise<ExtractedFaq[]> {
  // Prepare transcript excerpts (limit tokens)
  const transcriptData = calls
    .filter(c => c.transcript)
    .map(c => ({
      id: c.id,
      summary: c.summary,
      // Extract question portions from transcript
      questions: extractQuestionsFromTranscript(c.transcript)
    }))
    .slice(0, 50); // Limit to 50 calls
  
  const verticalContext: Record<string, string> = {
    TRADES: "plumbing, heating, electrical, or general trade services",
    VET: "veterinary care, pet health, and animal services",
    SALON: "hair styling, beauty treatments, and personal care",
    DENTAL: "dental care, oral health, and dental procedures",
    PHYSIO: "physiotherapy, rehabilitation, and physical health"
  };
  
  const response = await openai.chat.completions.create({
    model: "gpt-4o",
    messages: [
      {
        role: "system",
        content: `You are analysing customer call data for a ${verticalContext[vertical] || 'service'} business.

Your task is to identify the most common questions customers ask and generate helpful FAQ answers.

Rules:
1. Extract 8-12 FAQs maximum
2. Questions should be what real customers ask, not generic
3. Answers should be helpful, specific, and encourage contact
4. Categorise each: PRICING, AVAILABILITY, SERVICES, EMERGENCY, AREA, BOOKING, GENERAL
5. Include confidence score (0-1) based on how often this question appears
6. Answers should be 2-3 sentences, conversational but professional

Output JSON format:
{
  "faqs": [
    {
      "question": "Do you offer emergency callouts?",
      "answer": "Yes, we provide 24/7 emergency callouts for urgent issues. Call us anytime and we'll get someone to you as quickly as possible.",
      "category": "EMERGENCY",
      "confidence": 0.85
    }
  ]
}`
      },
      {
        role: "user",
        content: `Here are the call summaries and questions extracted from recent customer calls:\n\n${JSON.stringify(transcriptData, null, 2)}`
      }
    ],
    response_format: { type: "json_object" },
    temperature: 0.3
  });
  
  const result = JSON.parse(response.choices[0].message.content || '{"faqs":[]}');
  return result.faqs;
}

function extractQuestionsFromTranscript(transcript: string): string[] {
  // Simple extraction - find sentences ending in ?
  const questions = transcript.match(/[^.!?]*\?/g) || [];
  return questions.slice(0, 5); // Limit per call
}

function similarQuestion(q1: string, q2: string): boolean {
  // Simple similarity check - could use embeddings for better matching
  const normalize = (s: string) => s.toLowerCase().replace(/[^a-z0-9]/g, '');
  const n1 = normalize(q1);
  const n2 = normalize(q2);
  
  // Check if 70%+ of words match
  const words1 = new Set(n1.split(/\s+/));
  const words2 = new Set(n2.split(/\s+/));
  const intersection = [...words1].filter(w => words2.has(w));
  
  return intersection.length / Math.max(words1.size, words2.size) > 0.7;
}
```

### File
`src/trigger/geo/monitor-citations.ts`

### Trigger
- Cron: Daily at 6am

### Implementation

```typescript
import { schedules } from "@trigger.dev/sdk/v3";
import { prisma } from "@/lib/prisma";

// Perplexity has a public API we can use
const PERPLEXITY_API_KEY = process.env.PERPLEXITY_API_KEY;

export const monitorAiCitations = schedules.task({
  id: "monitor-ai-citations",
  cron: "0 6 * * *", // 6am daily
  run: async () => {
    const clients = await prisma.client.findMany({
      where: { status: 'ACTIVE' },
      select: {
        id: true,
        businessName: true,
        vertical: true,
        serviceAreas: true,
        coveredNumber: true
      }
    });
    
    let citationsFound = 0;
    
    for (const client of clients) {
      // Generate search queries likely to surface this business
      const queries = generateSearchQueries(client);
      
      for (const query of queries) {
        try {
          // Check Perplexity
          const perplexityResult = await checkPerplexity(query, client.businessName);
          
          if (perplexityResult.mentioned) {
            // Check if we already logged this citation recently
            const existing = await prisma.aiCitation.findFirst({
              where: {
                clientId: client.id,
                platform: 'PERPLEXITY',
                query,
                detectedAt: { gte: new Date(Date.now() - 7 * 24 * 60 * 60 * 1000) }
              }
            });
            
            if (!existing) {
              await prisma.aiCitation.create({
                data: {
                  clientId: client.id,
                  platform: 'PERPLEXITY',
                  query,
                  queryCategory: categorizeQuery(query),
                  citationType: perplexityResult.type,
                  position: perplexityResult.position,
                  snippet: perplexityResult.snippet,
                  sentiment: perplexityResult.sentiment
                }
              });
              citationsFound++;
            }
          }
          
          // Rate limit - don't hammer the API
          await new Promise(r => setTimeout(r, 2000));
          
        } catch (error) {
          console.error(`Citation check failed for query "${query}":`, error);
        }
      }
      
      // Update visibility score
      await updateVisibilityScore(client.id);
    }
    
    return { citationsFound };
  }
});

function generateSearchQueries(client: {
  businessName: string;
  vertical: string;
  serviceAreas: string[];
}): string[] {
  const queries: string[] = [];
  
  const verticalTerms: Record<string, string[]> = {
    TRADES: ['plumber', 'electrician', 'heating engineer', 'boiler repair'],
    VET: ['vet', 'veterinary', 'animal hospital', 'pet doctor'],
    SALON: ['hair salon', 'hairdresser', 'beauty salon', 'barber'],
    DENTAL: ['dentist', 'dental practice', 'dental clinic'],
    PHYSIO: ['physiotherapist', 'physio', 'physical therapy']
  };
  
  const terms = verticalTerms[client.vertical] || ['services'];
  const areas = client.serviceAreas.slice(0, 3);
  
  // Generate location-based queries
  for (const term of terms.slice(0, 2)) {
    for (const area of areas) {
      queries.push(`best ${term} in ${area}`);
      queries.push(`${term} near ${area}`);
      queries.push(`emergency ${term} ${area}`);
      queries.push(`recommended ${term} ${area}`);
    }
  }
  
  // Add business name query
  queries.push(client.businessName);
  
  return queries.slice(0, 10); // Limit to 10 queries per client
}

async function checkPerplexity(
  query: string,
  businessName: string
): Promise<{
  mentioned: boolean;
  type?: string;
  position?: number;
  snippet?: string;
  sentiment?: string;
}> {
  if (!PERPLEXITY_API_KEY) {
    return { mentioned: false };
  }
  
  const response = await fetch('https://api.perplexity.ai/chat/completions', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${PERPLEXITY_API_KEY}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      model: 'llama-3.1-sonar-small-128k-online',
      messages: [
        {
          role: 'user',
          content: query
        }
      ]
    })
  });
  
  if (!response.ok) {
    throw new Error(`Perplexity API error: ${response.status}`);
  }
  
  const data = await response.json();
  const answer = data.choices[0]?.message?.content || '';
  
  // Check if business is mentioned
  const businessNameLower = businessName.toLowerCase();
  const answerLower = answer.toLowerCase();
  
  if (answerLower.includes(businessNameLower)) {
    // Find position in list if it's a recommendation
    const lines = answer.split('\n');
    let position = null;
    
    for (let i = 0; i < lines.length; i++) {
      if (lines[i].toLowerCase().includes(businessNameLower)) {
        // Check if it's in a numbered list
        const match = lines[i].match(/^(\d+)/);
        if (match) {
          position = parseInt(match[1]);
        }
        break;
      }
    }
    
    // Extract snippet around business mention
    const snippetStart = Math.max(0, answerLower.indexOf(businessNameLower) - 100);
    const snippetEnd = Math.min(answer.length, answerLower.indexOf(businessNameLower) + businessName.length + 100);
    const snippet = answer.slice(snippetStart, snippetEnd);
    
    // Simple sentiment analysis
    const positiveWords = ['recommended', 'excellent', 'great', 'trusted', 'reliable', 'professional', 'best'];
    const negativeWords = ['avoid', 'poor', 'bad', 'complaints', 'issues'];
    
    let sentiment = 'NEUTRAL';
    if (positiveWords.some(w => snippet.toLowerCase().includes(w))) {
      sentiment = 'POSITIVE';
    } else if (negativeWords.some(w => snippet.toLowerCase().includes(w))) {
      sentiment = 'NEGATIVE';
    }
    
    return {
      mentioned: true,
      type: position ? 'LISTED' : 'MENTIONED',
      position,
      snippet,
      sentiment
    };
  }
  
  return { mentioned: false };
}

function categorizeQuery(query: string): string {
  const queryLower = query.toLowerCase();
  
  if (queryLower.includes('best') || queryLower.includes('recommended')) {
    return 'RECOMMENDATION';
  }
  if (queryLower.includes('near') || queryLower.includes('in ')) {
    return 'LOCAL_SERVICE';
  }
  if (queryLower.includes('vs') || queryLower.includes('compare')) {
    return 'COMPARISON';
  }
  if (queryLower.includes('emergency') || queryLower.includes('urgent')) {
    return 'EMERGENCY';
  }
  
  return 'GENERAL';
}

async function updateVisibilityScore(clientId: string): Promise<void> {
  const thirtyDaysAgo = new Date(Date.now() - 30 * 24 * 60 * 60 * 1000);
  
  // Get citation data
  const citations = await prisma.aiCitation.findMany({
    where: {
      clientId,
      detectedAt: { gte: thirtyDaysAgo }
    }
  });
  
  // Calculate score (0-100)
  let score = 0;
  
  // Base score from citation count (max 40 points)
  score += Math.min(40, citations.length * 5);
  
  // Bonus for positive sentiment (max 20 points)
  const positiveCitations = citations.filter(c => c.sentiment === 'POSITIVE').length;
  score += Math.min(20, positiveCitations * 4);
  
  // Bonus for being listed/recommended (max 20 points)
  const recommendations = citations.filter(c => 
    c.citationType === 'DIRECT_RECOMMENDATION' || c.citationType === 'LISTED'
  ).length;
  score += Math.min(20, recommendations * 5);
  
  // Bonus for platform diversity (max 20 points)
  const platforms = new Set(citations.map(c => c.platform));
  score += platforms.size * 5;
  
  // Cap at 100
  score = Math.min(100, score);
  
  // Update GEO page
  await prisma.geoPage.update({
    where: { clientId },
    data: {
      aiVisibilityScore: score,
      citationCount: citations.length,
      scoreLastCalculated: new Date()
```

```typescript
import { Client, Review } from "@prisma/client";

interface SchemaInput {
  client: Client & { owner?: { name: string } };
  faqs: Array<{ question: string; answer: string }>;
  services: Array<{ name: string; description: string }>;
  reviews: Review[];
}

export function generateSchemaMarkup(input: SchemaInput): object {
  const { client, faqs, services, reviews } = input;
  
  const baseUrl = `https://covered.ai/find/${slugify(client.businessName)}`;
  
  // Calculate aggregate rating
  const avgRating = reviews.length > 0
    ? reviews.reduce((sum, r) => sum + r.rating, 0) / reviews.length
    : null;
  
  const graph: object[] = [];
  
  // 1. LocalBusiness Schema
  const localBusiness: any = {
    "@type": getBusinessType(client.vertical),
    "@id": `${baseUrl}#business`,
    "name": client.businessName,
    "url": baseUrl,
    "telephone": client.coveredNumber,
    "email": client.email,
    "description": client.geoPage?.businessDescription || 
      `${client.businessName} provides professional ${client.vertical.toLowerCase()} services.`,
    "priceRange": "££", // Could be made dynamic
    "paymentAccepted": ["Cash", "Credit Card", "Bank Transfer"],
    "currenciesAccepted": "GBP"
  };
  
  // Add address if available
  if (client.address) {
    localBusiness.address = {
      "@type": "PostalAddress",
      "streetAddress": client.address,
      "addressLocality": client.city,
      "postalCode": client.postcode,
      "addressCountry": "GB"
    };
  }
  
  // Add geo coordinates if available
  if (client.lat && client.lng) {
    localBusiness.geo = {
      "@type": "GeoCoordinates",
      "latitude": client.lat,
      "longitude": client.lng
    };
  }
  
  // Add opening hours if available
  if (client.openingHours) {
    localBusiness.openingHoursSpecification = formatOpeningHours(client.openingHours);
  }
  
  // Add aggregate rating
  if (avgRating && reviews.length > 0) {
    localBusiness.aggregateRating = {
      "@type": "AggregateRating",
      "ratingValue": avgRating.toFixed(1),
      "bestRating": "5",
      "worstRating": "1",
      "reviewCount": reviews.length
    };
  }
  
  // Add area served
  if (client.serviceAreas?.length > 0) {
    localBusiness.areaServed = client.serviceAreas.map(area => ({
      "@type": "City",
      "name": area
    }));
  }
  
  // Add contact point
  localBusiness.contactPoint = {
    "@type": "ContactPoint",
    "telephone": client.coveredNumber,
    "contactType": "customer service",
    "availableLanguage": "English"
  };
  
  graph.push(localBusiness);
  
  // 2. FAQPage Schema
  if (faqs.length > 0) {
    graph.push({
      "@type": "FAQPage",
      "@id": `${baseUrl}#faq`,
      "mainEntity": faqs.map(faq => ({
        "@type": "Question",
        "name": faq.question,
        "acceptedAnswer": {
          "@type": "Answer",
          "text": faq.answer
        }
      }))
    });
  }
  
  // 3. Service Schemas
  services.forEach((service, index) => {
    graph.push({
      "@type": "Service",
      "@id": `${baseUrl}#service-${index}`,
      "name": service.name,
      "description": service.description,
      "provider": { "@id": `${baseUrl}#business` },
      "areaServed": client.serviceAreas?.map(area => ({
        "@type": "City",
        "name": area
      }))
    });
  });
  
  // 4. Review Schemas (sample of best reviews)
  const topReviews = reviews
    .filter(r => r.rating >= 4 && r.reviewText)
    .slice(0, 5);
  
  topReviews.forEach((review, index) => {
    graph.push({
      "@type": "Review",
      "@id": `${baseUrl}#review-${index}`,
      "itemReviewed": { "@id": `${baseUrl}#business` },
      "reviewRating": {
        "@type": "Rating",
        "ratingValue": review.rating,
        "bestRating": "5"
      },
      "reviewBody": review.reviewText,
      "datePublished": review.createdAt.toISOString().split('T')[0],
      "author": {
        "@type": "Person",
        "name": review.reviewerName || "Customer"
      }
    });
  });
  
  // 5. WebPage with Speakable
  graph.push({
    "@type": "WebPage",
    "@id": baseUrl,
    "name": `${client.businessName} - ${getVerticalTitle(client.vertical)}`,
    "description": localBusiness.description,
    "speakable": {
      "@type": "SpeakableSpecification",
      "cssSelector": [
        ".business-description",
        ".faq-answer",
        ".service-description"
      ]
    },
    "mainEntity": { "@id": `${baseUrl}#business` }
  });
  
  // 6. Organization (for additional trust signals)
  graph.push({
    "@type": "Organization",
    "@id": `${baseUrl}#organization`,
    "name": client.businessName,
    "url": baseUrl,
    "logo": client.logoUrl || undefined,
    "sameAs": [
      client.facebookUrl,
      client.instagramUrl,
      client.linkedinUrl
    ].filter(Boolean)
  });
  
  return {
    "@context": "https://schema.org",
    "@graph": graph
  };
}

function getBusinessType(vertical: string): string {
  const types: Record<string, string> = {
    TRADES: "HomeAndConstructionBusiness",
    VET: "VeterinaryCare",
    SALON: "BeautySalon",
    DENTAL: "Dentist",
    PHYSIO: "PhysicalTherapy",
    LEGAL: "LegalService",
    ACCOUNTING: "AccountingService"
  };
  return types[vertical] || "LocalBusiness";
}

function getVerticalTitle(vertical: string): string {
  const titles: Record<string, string> = {
    TRADES: "Professional Trade Services",
    VET: "Veterinary Practice",
    SALON: "Beauty & Hair Salon",
    DENTAL: "Dental Practice",
    PHYSIO: "Physiotherapy Clinic",
    LEGAL: "Legal Services",
    ACCOUNTING: "Accounting Services"
  };
  return titles[vertical] || "Professional Services";
}

function formatOpeningHours(hours: any): object[] {
  // Convert stored hours format to schema.org format
  const dayMapping: Record<string, string> = {
    monday: "Monday",
    tuesday: "Tuesday",
    wednesday: "Wednesday",
    thursday: "Thursday",
    friday: "Friday",
    saturday: "Saturday",
    sunday: "Sunday"
  };
  
  return Object.entries(hours)
    .filter(([_, value]: [string, any]) => value.open && value.close)
    .map(([day, times]: [string, any]) => ({
      "@type": "OpeningHoursSpecification",
      "dayOfWeek": dayMapping[day],
      "opens": times.open,
      "closes": times.close
    }));
}

function slugify(text: string): string {
  return text
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/(^-|-$)/g, '');
}
```
