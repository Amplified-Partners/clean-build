### MORNING STARTUP RITUAL (5 minutes)

**06:00 UTC / 06:00 AM Newcastle Time**

```bash
#!/bin/bash
# FILE: ~/scripts/morning-startup.sh

echo "🌅 KILO CODE MORNING STARTUP SEQUENCE"
echo "=================================="

# 1. Check system health
echo "✓ Checking Mac M4 system health..."
system_profiler SPHardwareDataType | grep "Memory" | awk '{print "Available RAM: " $4}'

# 2. Verify Kilo instances offline status
echo "✓ Verifying Kilo instances can boot..."
ps aux | grep "Kilo" | grep -v grep

# 3. Check API keys are loaded
echo "✓ Checking Perplexity API key..."
if [ -z "$PERPLEXITY_API_KEY" ]; then
  echo "⚠️ WARNING: PERPLEXITY_API_KEY not set. Loading..."
  export PERPLEXITY_API_KEY=$(cat ~/.secrets/perplexity.key)
fi

# 4. Check OpenRouter access
echo "✓ Checking OpenRouter API..."
curl -s https://openrouter.ai/api/v1/models -H "Authorization: Bearer $OPENROUTER_API_KEY" | jq '.data | length'

# 5. MLX local models ready?
echo "✓ Checking local MLX models..."
ls -la ~/.mlx/models/ | wc -l

# 6. Asset inventory quick scan
echo "✓ Running quick asset audit..."
./scripts/asset-audit.sh --quick

echo "✅ MORNING STARTUP COMPLETE"
```

## KILO CODE ORCHESTRATION MASTERY

### Understanding Orchestrator Mode

Orchestrator Mode is the **heart of scaling Kilo Code**. Instead of one monolithic task, you break complex projects into specialized subtasks.

**Basic Concept:**

```
Your Complex Goal
      ↓
  Orchestrator (Parent Task)
      ↓
  ┌───┴─────┬──────────┬──────────┐
  ↓         ↓          ↓          ↓
ARCHITECT  CODE      DEBUG    OPTIMIZER
(Plan)    (Build)   (Test)   (Polish)
  ↓         ↓          ↓          ↓
  └───┬─────┴──────────┴──────────┘
      ↓
  Integrated Solution
```

### Real-World Examples for Your Use Cases

**USE CASE 1: Build Automated Email Marketing System**

```yaml
ORCHESTRATOR TASK: Email Marketing Automation Platform

SUBTASK 1 - ARCHITECT MODE (Claude Sonnet)
└─ Input: "Design an email automation system that integrates with Mailchimp, 
           personalizes content using AI, tracks metrics, and scales to 10K 
           subscribers"
   Output: System architecture, API contracts, security model, database schema
   Estimated Time: 15 minutes

SUBTASK 2 - CODE MODE (Claude Sonnet)
└─ Input: Architecture from SUBTASK 1

   Task: "Build React frontend dashboard with campaign builder,
```

```python
    def route_task(self, task_description: str, task_type: str,

                   speed_critical: bool = False) -> Dict:
        """Route task to best instance"""
        
        # Match task type to instances
        candidates = []
        for name, config in self.instances.items():
            if task_type in config["best_for"]:
                candidates.append((name, config))
        
        if not candidates:
            candidates = list(self.instances.items())
        
        # Score candidates
        scores = []
        for name, config in candidates:
            score = 0
            
            # Cost factor
            if cost_sensitive:
                score += (1 - (config["cost_per_1k"] / 5.0)) * 0.4  # Cheaper = higher score
            else:
                score += 0.1  # Neutral cost weight
            
            # Speed factor
            if speed_critical:
                score += (config["speed_tokens_per_sec"] / 300) * 0.4  # Faster = higher score
            else:
                score += 0.1  # Neutral speed weight
            
            # Availability
            if self.is_available(name):
                score += 0.2
            else:
                score += 0  # Penalize unavailable
            
            # Budget check
            if config["cost_per_1k"] > 0 and (self.daily_spent + config["cost_per_1k"]/1000 > self.daily_budget):
                score *= 0.3  # Reduce score if over budget
            
            scores.append((name, config, score))
        
        # Select best
        best_instance, best_config, best_score = max(scores, key=lambda x: x[2])
        
        return {
            "selected_instance": best_instance,
            "model": best_config["model"],
            "url": best_config["url"],

                return False
        return True
```

```python
    def call_with_fallback(self,

        Args:
            prompt: The prompt to send
            task_type: Type of task (system_design, code_gen, debugging, etc.)
            max_tokens: Max completion tokens
            temperature: Sampling temperature
            preferred_provider: If specified, try this first
        
        Returns:

                self._log_call(provider_name, response, prompt)

                    "success": True,
                    "provider": provider_name,
                    "content": response.get("content"),
                    "usage": response.get("usage"),

            "code_generation": ["anthropic", "openrouter", "local_mlx"],
            "debugging": ["anthropic", "openrouter", "local_mlx"],
            "quick_analysis": ["openrouter", "local_mlx", "anthropic"],  # Speed-first
            "cost_sensitive": ["local_mlx", "openrouter", "anthropic"],  # Cheap first
            "general": ["anthropic", "openrouter", "perplexity", "local_mlx"],
        }
        
        chain = chains.get(task_type, chains["general"])
        
        # If provider preferred, move it to front
        if preferred and preferred in chain:

            chain.insert(0, preferred)

                }

            # Direct Anthropic API
            response = requests.post(

                    "temperature": temperature,
                    "messages": [{"role": "user", "content": prompt}]
                }
            )
            response_json = response.json()
            return {
                "content": response_json["content"][0]["text"],
                "usage": {
                    "input_tokens": response_json["usage"]["input_tokens"],
                    "output_tokens": response_json["usage"]["output_tokens"]
                }
            }
        
        elif provider == "openrouter":
            # OpenRouter (unified interface)
            response = requests.post(
                f"{config['base_url']}/chat/completions",
                headers={"Authorization": f"Bearer {config['api_key']}"},
                json={
                    "model": "anthropic/claude-3.5-sonnet",
                    "messages": [{"role": "user", "content": prompt}],
                    "max_tokens": max_tokens,
                    "temperature": temperature
                }
            )

    def _check_budget_exceeded(self, provider: str) -> bool:
        """Check if provider budget exceeded for today"""
        return self.daily_spent.get(provider, 0.0) >= self.daily_budget.get(provider, float('inf'))
    
    def _calculate_cost(self, provider: str, usage: Dict) -> float:
        """Calculate API call cost in USD"""
        config = self.providers[provider]
        
        input_cost = (usage.get("input_tokens", 0) / 1_000_000) * config["cost_per_1m_input"]
        output_cost = (usage.get("output_tokens", 0) / 1_000_000) * config["cost_per_1m_output"]
        total_cost = input_cost + output_cost
        
        # Track daily spend
        if provider in self.daily_spent:
            self.daily_spent[provider] += total_cost
        
        return total_cost

        """Log API call for auditing"""

            "provider": provider,
            "prompt_length": len(prompt),
            "usage": response.get("usage", {}),
            "cost": self._calculate_cost(provider, response.get("usage", {}))
        })
```
