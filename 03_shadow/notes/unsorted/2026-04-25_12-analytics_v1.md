# Module 12: Analytics & Reporting

## Overview
Track everything. Calls, leads, conversions, revenue, churn. Internal dashboard for Ewan, client-facing reports for customers.

## Expert Panel Input

**Data Expert**: Track the metrics that matter for decisions. Vanity metrics are noise.

**Visualization Expert**: Simple charts, clear trends, actionable insights. No dashboards that require training.

**Business Intelligence Expert**: Cohort analysis, LTV predictions, churn indicators. Know the health of each customer.

---

## Key Metrics

### Business Health
- MRR (Monthly Recurring Revenue)
- ARR (Annual Recurring Revenue)
- Customer count
- Churn rate
- LTV (Lifetime Value)
- CAC (Customer Acquisition Cost)

### Operations
- Total calls handled
- Avg call duration
- Lead capture rate
- Emergency forward rate
- Gemma uptime

### Customer Health
- Calls per customer
- Days since last call
- Feature usage
- Support tickets
- At-risk indicators

"""
Analytics Tracker

Tracks all events and generates reports.
"""

import os
from typing import Optional, List, Dict, Any
from dataclasses import dataclass, field
from datetime import datetime, timedelta
from enum import Enum
from collections import defaultdict
import json


class EventType(str, Enum):
    # Acquisition
    SIGNUP_STARTED = "signup_started"
    SIGNUP_COMPLETED = "signup_completed"
    TRIAL_STARTED = "trial_started"
    TRIAL_CONVERTED = "trial_converted"
    
    # Activation
    FIRST_CALL = "first_call"
    FORWARDING_SETUP = "forwarding_setup"
    DASHBOARD_LOGIN = "dashboard_login"
    
    # Revenue
    PAYMENT_RECEIVED = "payment_received"
    PAYMENT_FAILED = "payment_failed"
    SUBSCRIPTION_CANCELLED = "subscription_cancelled"
    SUBSCRIPTION_REACTIVATED = "subscription_reactivated"
    
    # Engagement
    CALL_RECEIVED = "call_received"
    LEAD_CAPTURED = "lead_captured"
    EMERGENCY_FORWARDED = "emergency_forwarded"
    SETTINGS_UPDATED = "settings_updated"
    
    # Support
    SUPPORT_TICKET = "support_ticket"
    SUPPORT_RESOLVED = "support_resolved"


@dataclass
class Event:
    id: str
    event_type: EventType
    client_id: Optional[str]
    properties: Dict[str, Any]
    timestamp: datetime = field(default_factory=datetime.utcnow)


@dataclass
class ClientHealth:
    client_id: str
    health_score: float  # 0-100
    risk_level: str  # low, medium, high
    signals: List[str]
    last_call_at: Optional[datetime]
    calls_last_30_days: int
    days_active: int
    mrr: float


class AnalyticsTracker:
    """
    Tracks events and generates analytics.
    """
    
    def __init__(self):
        self.events: List[Event] = []
        self.client_metrics: Dict[str, Dict] = {}
    
    async def track(
        self,
        event_type: EventType,
        client_id: Optional[str] = None,
        properties: Optional[Dict] = None,
    ) -> Event:
        """Track an event."""
        import uuid
        
        event = Event(
            id=str(uuid.uuid4()),
            event_type=event_type,
            client_id=client_id,
            properties=properties or {},
        )
        
        self.events.append(event)
        
        # Update client metrics
        if client_id:
            self._update_client_metrics(client_id, event)
        
        return event
    
    def _update_client_metrics(self, client_id: str, event: Event):
        """Update rolling metrics for a client."""
        if client_id not in self.client_metrics:
            self.client_metrics[client_id] = {
                "first_event_at": event.timestamp,
                "last_event_at": event.timestamp,
                "total_events": 0,
                "calls": 0,
                "leads": 0,
                "payments": 0,
                "total_revenue": 0.0,
            }
        
        metrics = self.client_metrics[client_id]
        metrics["last_event_at"] = event.timestamp
        metrics["total_events"] += 1
        
        if event.event_type == EventType.CALL_RECEIVED:
            metrics["calls"] += 1
        elif event.event_type == EventType.LEAD_CAPTURED:
            metrics["leads"] += 1
        elif event.event_type == EventType.PAYMENT_RECEIVED:
            metrics["payments"] += 1
            metrics["total_revenue"] += event.properties.get("amount", 0)
    
    async def get_business_metrics(
        self,
        start_date: Optional[datetime] = None,
        end_date: Optional[datetime] = None,
    ) -> Dict[str, Any]:
        """Get high-level business metrics."""
        if not start_date:
            start_date = datetime.utcnow() - timedelta(days=30)
        if not end_date:
            end_date = datetime.utcnow()
        
        period_events = [
            e for e in self.events
            if start_date <= e.timestamp <= end_date
        ]
        
        # Calculate MRR (simplified)
        payment_events = [
            e for e in period_events
            if e.event_type == EventType.PAYMENT_RECEIVED
        ]
        total_revenue = sum(e.properties.get("amount", 0) for e in payment_events)
        
        # Count customers
        active_clients = set(
            e.client_id for e in period_events
            if e.client_id and e.event_type == EventType.CALL_RECEIVED
        )
        
        # Signups
        signups = len([
            e for e in period_events
            if e.event_type == EventType.SIGNUP_COMPLETED
        ])
        
        # Churn
        churned = len([
            e for e in period_events
            if e.event_type == EventType.SUBSCRIPTION_CANCELLED
        ])
        
        return {
            "period": {
                "start": start_date.isoformat(),
                "end": end_date.isoformat(),
            },
            "revenue": {
                "total": total_revenue,
                "mrr": total_revenue,  # Simplified
                "arr": total_revenue * 12,
            },
            "customers": {
                "active": len(active_clients),
                "new": signups,
                "churned": churned,
                "net_change": signups - churned,
            },
            "calls": {
                "total": len([e for e in period_events if e.event_type == EventType.CALL_RECEIVED]),
                "leads": len([e for e in period_events if e.event_type == EventType.LEAD_CAPTURED]),
                "emergencies": len([e for e in period_events if e.event_type == EventType.EMERGENCY_FORWARDED]),
            },
        }
    
    async def get_client_health(self, client_id: str) -> ClientHealth:
        """Calculate health score for a client."""
        metrics = self.client_metrics.get(client_id, {})
        
        signals = []
        score = 100.0
        
        # Check for recent activity
        last_event = metrics.get("last_event_at")
        if last_event:
            days_since = (datetime.utcnow() - last_event).days
            if days_since > 14:
                signals.append("No activity in 14+ days")
                score -= 30
            elif days_since > 7:
                signals.append("No activity in 7+ days")
                score -= 15
        else:
            signals.append("No events recorded")
            score -= 50
        
        # Check call volume
        calls_30d = len([
            e for e in self.events
            if e.client_id == client_id
            and e.event_type == EventType.CALL_RECEIVED
            and e.timestamp > datetime.utcnow() - timedelta(days=30)
        ])
        
        if calls_30d == 0:
            signals.append("No calls in 30 days")
            score -= 25
        elif calls_30d < 5:
            signals.append("Low call volume (<5/month)")
            score -= 10
        
        # Check payment status
        recent_payments = [
            e for e in self.events
            if e.client_id == client_id
            and e.event_type == EventType.PAYMENT_RECEIVED
            and e.timestamp > datetime.utcnow() - timedelta(days=35)
        ]
        
        if not recent_payments:
            signals.append("No recent payment")
            score -= 20
        
        # Determine risk level
        if score >= 70:
            risk = "low"
        elif score >= 40:
            risk = "medium"
        else:
            risk = "high"
        
        # Get last call
        call_events = [
            e for e in self.events
            if e.client_id == client_id and e.event_type == EventType.CALL_RECEIVED
        ]
        last_call = max((e.timestamp for e in call_events), default=None)
        
        return ClientHealth(
            client_id=client_id,
            health_score=max(0, score),
            risk_level=risk,
            signals=signals,
            last_call_at=last_call,
            calls_last_30_days=calls_30d,
            days_active=(datetime.utcnow() - metrics.get("first_event_at", datetime.utcnow())).days,
            mrr=297.0,  # TODO: Get from subscription
        )
    
    async def get_cohort_analysis(self) -> Dict[str, Any]:
        """Analyze customer cohorts by signup month."""
        cohorts = defaultdict(lambda: {
            "signups": 0,
            "active_month_1": 0,
            "active_month_2": 0,
            "active_month_3": 0,
            "churned": 0,
        })
        
        # Group signups by month
        for event in self.events:
            if event.event_type == EventType.SIGNUP_COMPLETED and event.client_id:
                cohort_key = event.timestamp.strftime("%Y-%m")
                cohorts[cohort_key]["signups"] += 1
        
        return dict(cohorts)
    
    async def get_client_report(
        self,
        client_id: str,
        start_date: Optional[datetime] = None,
        end_date: Optional[datetime] = None,
    ) -> Dict[str, Any]:
        """Generate a report for a specific client."""
        if not start_date:
            start_date = datetime.utcnow() - timedelta(days=30)
        if not end_date:
            end_date = datetime.utcnow()
        
        client_events = [
            e for e in self.events
            if e.client_id == client_id
            and start_date <= e.timestamp <= end_date
        ]
        
        calls = [e for e in client_events if e.event_type == EventType.CALL_RECEIVED]
        leads = [e for e in client_events if e.event_type == EventType.LEAD_CAPTURED]
        emergencies = [e for e in client_events if e.event_type == EventType.EMERGENCY_FORWARDED]
        
        # Daily breakdown
        daily = defaultdict(lambda: {"calls": 0, "leads": 0})
        for call in calls:
            day = call.timestamp.strftime("%Y-%m-%d")
            daily[day]["calls"] += 1
        for lead in leads:
            day = lead.timestamp.strftime("%Y-%m-%d")
            daily[day]["leads"] += 1
        
        # Hourly distribution
        hourly = defaultdict(int)
        for call in calls:
            hourly[call.timestamp.hour] += 1
        
        return {
            "period": {
                "start": start_date.isoformat(),
                "end": end_date.isoformat(),
            },
            "summary": {
                "total_calls": len(calls),
                "leads_captured": len(leads),
                "emergencies_forwarded": len(emergencies),
                "lead_rate": len(leads) / len(calls) * 100 if calls else 0,
            },
            "daily": dict(daily),
            "hourly_distribution": dict(hourly),
            "busiest_hour": max(hourly, key=hourly.get) if hourly else None,
            "avg_calls_per_day": len(calls) / max(1, (end_date - start_date).days),
        }


# Singleton
analytics = AnalyticsTracker()

"""
Analytics API Routes
"""

from fastapi import APIRouter, HTTPException
from typing import Optional
from datetime import datetime, timedelta

from src.services.analytics.tracker import analytics, EventType

router = APIRouter(prefix="/analytics", tags=["analytics"])


@router.get("/business")
async def get_business_metrics(
    days: int = 30,
):
    """Get business-level metrics."""
    start_date = datetime.utcnow() - timedelta(days=days)
    return await analytics.get_business_metrics(start_date=start_date)


@router.get("/clients/{client_id}/health")
async def get_client_health(client_id: str):
    """Get health score for a client."""
    return await analytics.get_client_health(client_id)


@router.get("/clients/{client_id}/report")
async def get_client_report(
    client_id: str,
    days: int = 30,
):
    """Get detailed report for a client."""
    start_date = datetime.utcnow() - timedelta(days=days)
    return await analytics.get_client_report(client_id, start_date=start_date)


@router.get("/cohorts")
async def get_cohort_analysis():
    """Get cohort retention analysis."""
    return await analytics.get_cohort_analysis()


@router.get("/at-risk")
async def get_at_risk_clients():
    """Get list of at-risk clients."""
    at_risk = []
    
    for client_id in analytics.client_metrics.keys():
        health = await analytics.get_client_health(client_id)
        if health.risk_level in ["medium", "high"]:
            at_risk.append({
                "client_id": client_id,
                "health_score": health.health_score,
                "risk_level": health.risk_level,
                "signals": health.signals,
                "last_call": health.last_call_at.isoformat() if health.last_call_at else None,
            })
    
    return sorted(at_risk, key=lambda x: x["health_score"])


@router.post("/track")
async def track_event(
    event_type: str,
    client_id: Optional[str] = None,
    properties: Optional[dict] = None,
):
    """Track a custom event."""
    try:
        et = EventType(event_type)
    except ValueError:
        raise HTTPException(status_code=400, detail="Invalid event type")
    
    event = await analytics.track(et, client_id, properties)
    return {"event_id": event.id}

"use client";

import React, { useEffect, useState } from "react";
import { 
  TrendingUp,

interface BusinessMetrics {
  revenue: {
    total: number;
    mrr: number;
    arr: number;
  };
  customers: {
    active: number;
    new: number;
    churned: number;
    net_change: number;
  };
  calls: {
    total: number;
    leads: number;
    emergencies: number;
  };
}

interface AtRiskClient {
  client_id: string;
  health_score: number;
  risk_level: string;
  signals: string[];
  last_call: string | null;
}

        setLoading(false);
      }

  if (loading) {
    return <div className="p-8">Loading analytics...</div>;
  }

  return (
    <div className="space-y-6">
      <div className="flex items-center justify-between">
        <h1 className="text-2xl font-bold">Analytics</h1>
        <select
          value={period}

      </div>

      {/* Revenue */}
      <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
        <MetricCard
          icon={DollarSign}
          label="MRR"

        />
        <MetricCard
          icon={DollarSign}
          label="ARR"
          value={`£${metrics?.revenue.arr.toLocaleString() || 0}`}
          color="green"
        />
        <MetricCard
          icon={TrendingUp}
          label="Revenue (Period)"
          value={`£${metrics?.revenue.total.toLocaleString() || 0}`}
          color="blue"
        />
      </div>

      {/* Customers */}

function MetricCard({
  icon: Icon,
  label,
  value,
  color,
}: {
  icon: any;
  label: string;
  value: string | number;
  color: string;
}) {
  const colorClasses: Record<string, string> = {
    blue: "bg-blue-50 text-blue-600",
    green: "bg-green-50 text-green-600",
    red: "bg-red-50 text-red-600",
    purple: "bg-purple-50 text-purple-600",
    orange: "bg-orange-50 text-orange-600",
  };

  return (
    <div className="bg-white rounded-xl shadow-sm p-4">
      <div className="flex items-center gap-3 mb-2">
        <div className={`p-2 rounded-lg ${colorClasses[color]}`}>
          <Icon className="w-4 h-4" />
        </div>
        <span className="text-sm text-gray-500">{label}</span>
      </div>
      <p className="text-2xl font-bold">{value}</p>
    </div>
  );
}
